#!/bin/sh


# All TCP sessions should begin with SYN
iptables -A INPUT -p 22 -p 17 ! --dport 4  -j DROP
iptables -A INPUT -p 22 -p 6 ! --dport 4  -j DROP
iptables -A INPUT -p 22 ! -d 71.82.93.102/32  -j DROP
iptables -A INPUT -p 22 -d 71.82.93.102/32 -p 17 --dport 4 -j ACCEPT
iptables -A INPUT -p 22 -d 71.82.93.102/32 -p 6 --dport 4 -j ACCEPT
iptables -A INPUT -p tcp ! --syn -m state --state NEW -j DROP
iptables -A INPUT -j DROP
